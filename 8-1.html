function has(list, item){
  return list.includes(item)
}
function union(list, item){
  if(has(list, item)){
    return list
  }
  return [...list, item]
}
module.exports = union
        

const union = require('./union.js')
console.log(union([1, 2], 3))
// [1, 2, 3]
console.log(union([1, 2], 2))
// [1, 2]
