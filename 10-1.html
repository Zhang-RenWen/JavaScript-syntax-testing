現代 Javascript 實務應用

新標準 ES2016 (ES7)

10-1 Array.prototype.includes()

ES7 之前我們判斷一個陣列裡面是否包含指定元素 


let arr = ['JOJO', 'Joseph Joestar', 'Caesar', 'Kars'];
if (arr.indexOf('JOJO') !== -1) {
  console.log('Hi~ JOJO');
}

ES7 之後可以寫成這樣,會更加直覺‧ 


let arr = ['JOJO', 'Joseph Joestar', 'Caesar', 'Kars'];
if (arr.includes('JOJO')) {  // ES7 Array.prototype.includes()
  console.log('Hi~ JOJO');
}

實際應用做條件判斷部分情境下可以代換 if..else 


let conditionsA =[(()=>{return true;})(),false,(()=>{return false;})()]
console.log(conditionsA.includes(true))  // true
let conditionsB = [(()=>{return false;})(),false,(()=>{return false;})()]
console.log(conditionsB.includes(true))  // false
        

console.log([NaN].includes(NaN));  // true
console.log([null].includes(null));  // true
console.log([undefined].includes(undefined)); // true

// 物件因為不同參考故無法做比對
// 若要比對必須指向同一參考變數。

console.log([{a:1}].includes({a:1})); // false
const obj = {a:1}
console.log([obj].includes(obj)); // true

// 萬國碼比對
const array = ['🥗', '🍔', '🍰'];
array.includes('🍰'); // true


// Symbol 比對
const symbol = Symbol('symbol');
const array = [symbol];
array.includes(symbol); // true
[Symbol('symbol')].includes(Symbol('symbol')) // false
