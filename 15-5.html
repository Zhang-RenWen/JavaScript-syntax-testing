15-5 Logical AND assignment(&&=)


let a = 1;
let b = 0;

a &&= 2;
console.log(a);
// Expected output: 2

b &&= 2;
console.log(b);
// Expected output: 0

        

x &&= y
        

x && (x = y);
        

const x = 0;
x &&= 2;
        

const x = {
  get value() {
    return 0;
  },
  set value(v) {
    console.log("Setter called");
  },
};

x.value &&= 2;
        

const x = 0;
x &&= console.log("y evaluated");
// Logs nothing
        

let x = 0;
let y = 1;

x &&= 0; // 0
x &&= 1; // 0
y &&= 1; // 1
y &&= 0; // 0
