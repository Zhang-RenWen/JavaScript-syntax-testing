14-5 動態 import

ES6 時候的 import 屬於靜態載入。ES11 以後可以實現按需載入。


// hello.js
export function hello(){
  alert('Hello')
}

// app.js
// import * as m1 from "./hello.js" // 靜態加載  ;

const btn = document.getElementById('btn');
btn.onclick = function(){
  import('./hello.js').then(module => { // 懶加載
  module.hello(); // Hello
});
}
