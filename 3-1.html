class LocalStorage {
  constructor(key) {
    this.key = key
  }
  
  get data() {
    return JSON.parse(localStorage.getItem(this.key))
  }
  
  set data(data) {
    localStorage.setItem(this.key, JSON.stringify(data))
  }
}
  

const ls = new LocalStorage('groceries')
ls.data = ['apples', 'bananas', 'grapes']
