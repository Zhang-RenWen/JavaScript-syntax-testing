2-1-1 特性值指定簡化語法


  // before ES6
  var listeners = {}
  function run(){}
  var events = { 
listeners: listeners,
run: run 
  }

  let listeners = {}
  function run(){}
  let events = { listeners , run }

2-1-3 方法的定義


// before ES6
    
var emitter = {
  events:{},
  on:function(type,fn){
    if(this.events[type]===undefined){
      this.events[type]=[]
    }
    this.events[type].push(fn)
  },
  emit:function(type,event){
    if(this.events[type]===undefined){
      return 
    }
    this.events[type].forEach(function(fn){
      fn(event)
    })
  }
}

// after ES6
let emitter = {
  events:{},
  on(type,fn){
    if(this.events[type]===undefined){
      this.events[type]=[]
    }
    this.events[type].push(fn)
  },
  emit(type,event){
    if(this.events[type]===undefined){
      return 
    }
    this.events[type].forEach(function(fn){
      fn(event)
    })
  }
}
