11-3 Trailing Commas

Array


 [
    'foo',
+   'baz',
    'bar',
-   'baz',
 ]
        

const arr = [
  1,
  2,
  3,
];
arr; // [1, 2, 3]
arr.length; // 3
      

const arr = [1, 2, 3, , ,];
arr.length; // 5

Objects


const object = {
  foo: 'bar',
  baz: 'qwerty',
  age: 42,
};
      

function f(p) {}
function f(p,) {}

(p) => {};
(p,) => {};
      

class C {
  one(a,) {}
  two(a, b,) {}
}

const obj = {
  one(a,) {},
  two(a, b,) {},
};

Function calls


f(p);
f(p,);

Math.max(10, 20);
Math.max(10, 20,);
