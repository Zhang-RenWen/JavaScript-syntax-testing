5-3 ES6 的集合


const set = new Set()
set.add({ an: 'example' })
        

const set = new Set(['a','b','c'])
console.log([...set])
// ['a','b','c'] 
        

const set = new Set(['a','b','b','c','c'])
console.log([...set])
// ['a','b','c'] 
        

function allDiv(){
  return document.querySelectorAll('div')
}
const set = new Set(allDiv()) 
console.log(set.size) // 56
allDiv().forEach(div=> set.add(div))
console.log(set.size) // 56
        

const set = new Set(['a','b','c'])
console.log([...set.entries()])
// [['a','a'],['b', 'b'],['c','c']]
