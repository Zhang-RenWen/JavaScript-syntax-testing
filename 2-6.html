2-6-4 Const 與 let 敘述的優點


var items = ['a', 'b', 'c']
var todo = checklist(items)
todo.check()
console.log(items)
// ['b', 'c']
items = ['d', 'e']
todo.check()
console.log(items)
// ['d', 'e'], would be ['c'] if items had been constant
function checklist(items) {
return {
  check: () => items.shift()
}
 }
